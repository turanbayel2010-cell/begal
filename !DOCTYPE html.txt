let xoTurn = "X";
function xoMove(cell){

            document.getElementById("xo-info").innerHTML = "–ü–æ–±–µ–¥–∏–ª: "+c[w[0]];

function memStart(){
    let field = document.getElementById("mem-field");
    field.innerHTML = "";
    let cards = ["A","A","B","B","C","C","D","D","E","E","F","F"];
    cards.sort(()=>Math.random()-0.5);

    let opened = [];

    cards.forEach(c=>{
        let d = document.createElement("div");
        d.style.width="90px";
        d.style.height="90px";
        d.style.border="1px solid black";
        d.style.margin="5px";
        d.style.fontSize="40px";
        d.style.display="flex";
        d.style.justifyContent="center";
        d.style.alignItems="center";
        d.dataset.v = c;
        d.onclick = ()=>{
            if(d.innerHTML) return;
            d.innerHTML = c;
            opened.push(d);
            if(opened.length===2){
                if(opened[0].dataset.v!==opened[1].dataset.v){
                    setTimeout(()=>{
                        opened[0].innerHTML="";
                        opened[1].innerHTML="";
                        opened=[];
                    },600);
                } else opened=[];
            }
        };
        field.appendChild(d);
    });
}

/* === 3. –ó–º–µ–π–∫–∞ === */
let snakeLoop;
function snakeStart(){
    let cvs = document.getElementById("snake");
    let ctx = cvs.getContext("2d");
    let box = 15;
    let snake = [{x:5,y:5}];
    let food = {x:10,y:10};
    let dx=1, dy=0;

    document.onkeydown = e=>{
        if(e.key==="ArrowUp" && dy===0){dx=0;dy=-1;}
        if(e.key==="ArrowDown" && dy===0){dx=0;dy=1;}
        if(e.key==="ArrowLeft" && dx===0){dx=-1;dy=0;}
        if(e.key==="ArrowRight" && dx===0){dx=1;dy=0;}
    };

    clearInterval(snakeLoop);
    snakeLoop = setInterval(()=>{
        let head = {x:snake[0].x+dx, y:snake[0].y+dy};

        if(head.x===food.x && head.y===food.y){
            food.x = Math.floor(Math.random()*30);
            food.y = Math.floor(Math.random()*30);
        } else snake.pop();

        snake.unshift(head);

        ctx.fillStyle="#eee";
        ctx.fillRect(0,0,450,450);

        ctx.fillStyle="green";
        snake.forEach(p=>ctx.fillRect(p.x*box,p.y*box,box,box));

        ctx.fillStyle="red";
        ctx.fillRect(food.x*box,food.y*box,box,box);
    },120);
}

let catchLoop;
function catchStart(){
    let cvs = document.getElementById("catcher");
    let ctx = cvs.getContext("2d");

    let player = {x:200,y:500,w:70,h:20};
    let obj = {x:Math.random()*380, y:0};

    document.onmousemove = e=>{
        player.x = e.clientX - 260; // –ø–æ–¥—Ö–æ–¥–∏—Ç –¥–ª—è –Ω–æ—É—Ç–±—É–∫–∞
    };

    clearInterval(catchLoop);
    catchLoop = setInterval(()=>{
        ctx.clearRect(0,0,450,550);

        obj.y += 5;
        if(obj.y > 550){
            obj.y = 0;
            obj.x = Math.random()*380;
        }

        ctx.fillStyle="blue";
        ctx.fillRect(player.x,player.y,player.w,player.h);

        ctx.fillStyle="red";
        ctx.fillRect(obj.x,obj.y,25,25);

        if(obj.x > player.x && obj.x < player.x+player.w &&
           obj.y > player.y && obj.y < player.y+player.h){
            obj.y = 0;
            obj.x = Math.random()*380;
        }
    },30);
}
